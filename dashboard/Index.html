cd ~/.openclaw/workspace/skills/mktagent001/dashboard
python3 server.py --host 0.0.0.0 --port 8765 --token your-token &
3. Cloudflare tunnel
pkill cloudflared && cloudflared tunnel --url http://localhost:8080 > /tmp/cloudflared.log 2>&1 &
sleep 5 && cat /tmp/cloudflared.log | grep trycloudflare
4. Follow logs
openclaw logs --follow
